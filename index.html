const submitData = async (event) => {
  event.preventDefault();

  // 1. Preparar el Payload Seguro
  const formData = new FormData(event.target);
  
  // Configuración Requerida
  formData.append('to', 'joshua44benja@gmail.com'); // <-- Tu Email

  try {
    // 2. Envío a la API Python/FastAPI
    const response = await fetch('https://forms-nicpages.vercel.app/submit-form', {
      method: 'POST',
      body: formData // El navegador maneja los headers y boundary
    });

    if (response.ok) {
       alert('¡Datos encriptados y enviados!');
    }
  } catch (error) {
    console.error('Error en el túnel:', error);
  }
};
