src/api/submit.js 
const submitForm = async (event) => {
  event.preventDefault();

  // 1. Capturar Datos
  const formData = new FormData(event.target);
  formData.append('to', 'joshua44benja@gmail.com');

  try {
    // 2. Envío Seguro
    const req = await fetch('https://forms-nicpages.vercel.app/submit-form', {
      method: 'POST',
      body: formData
    });

    if (req.ok) alert('¡Mensaje Enviado!');
  } catch (err) {
    console.error(err);
  }
};
